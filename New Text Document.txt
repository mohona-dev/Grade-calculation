    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
     double math ,english, science, tot,avg ;
     String grade;
     math = Integer.parseInt(txtmath.getText());
    english =Integer.parseInt(txteng.getText());
    science=Integer.parseInt(txesci.getText());
    
   tot = math+english+science;
   avg = tot/3;
   
   if(avg>=80){
    grade="A+";
}
   else if(avg>=70){
    grade="A";
}
     else if(avg>=60){
    grade="B";
}
   else if(avg>=50){
    grade="C";
} 
    else if(avg>=40){
    grade="D";
}
    else{
    grade="F";
}
    lbltot.setText(String.valueOf(tot));
    lblavg.setText(String.valueOf(avg));
    lblgrd.setText(grade);   // TODO add your handling code here:
    }
